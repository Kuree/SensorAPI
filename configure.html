ZeroMQ Queue

[ZeroMQQueueServer]
# Front end port defines where the queue should listen to, i.e. the port used by clients to send and receive data to.
FrontEndPort = 5555

# Back end port defines where the queue should listen to, i.e. the port used by workers to send and receive data to.
BackendPort = 5556

[Worker]
# How many liveness can a work has. After heart beating failure for several times (3 here), the queue will mark the worker as dead.
HeartBeatLiveness = 3

; Time interval for heart beating in seconds.
HeartBeatInterval = 1.0

ZeroMQ Worker

[ZeroMQQueueServer]
# The ip address of queue to talk to, the defualt is localhost
ServerIP = localhost

# the port of quque to talk to, the default is 5556
ServerPort = 5556

[Worker]
# Heart beat initial value
IntervalInit = 1

# Headt beat max value, i.e., after that, the queue is treated as dead.
IntervalMax = 32


[OpenTSDB]
# The ip address of OpenTSDB server
ServerIP = localhost

; The port of OpenTSDB server
ServerPort = 4242

ZeroMQ Client

[ZeroMQQueueServer]

# The quque's ip address of client to talk to, the defualt is localhost
ServerIP = localhost

# the queue's port of client to talk to, the default is 5555
ServerPort = 5555

[ZeroMQClient]
# How much time it should be treated as request timeout, in ms.
RequestTimeout = 100000

# If set true, then when the client puts data into the system, it will receive debug information.
isDebug = True
